home.ts

export class HomePage { 

  for (var i = 0 ; i < 10; i++) { 
    console.log(i); //Output 0-9 
  } 
  console.log(i); // Outputs 10

  for (let i = 0 ; i < 10; i++) { 
    console.log(i); //Outputs 0-9 
  } 
  console.log(i); // Uncaught ReferenceError: i is not defined

}
  • var: global scope
  • let: local scope

    greetingPromise.then(
      function (greeting) { 
        console.log(greeting); // 'Hello in the United States’ 
      }, 
      function (error) { 
        console.error('uh oh: ', error); // 'Drat!’ 
      }
    );

  }
}
  • Object: deferred/asynchronous computations

  • then: registers callbacks
  • Observables: resolves values asynchronously

  Arrow Functions

home.ts


export class HomePage { 
  constructor() {
  }
  func() {
    var multiply = function(x, y) { 
      return x * y; 
    };
    var multiply = (x, y) => { 
      return x * y;
    };
    var missions = [ 
      { name:'Mercury', flights:6 }, 
      { name:'Gemini', flights:10 }
    ];
    // ES5 
    console.log(missions.map( 
        function(mission) { 
          return mission.flights; 
        } 
    )); // [6, 10]
    // ES6
    console.log(missions.map( 
        mission => mission.flights 
    )); // [6, 10] 
  }
}
  • no function keyword
  • array manipulations

  Special Types

home.ts


export class HomePage { 
  constructor() {
  }
  func() {
    let someVar: any; 
    // Takes any and all variable types 
    someVar = '123'; 
    someVar = 123;

    // The null and undefined JavaScript literals are effectively treated as the same as the any type:
    var str: string; 
    var num: number; 
    // These literals can be assigned to anything 
    str = undefined; 
    num = null; 
  }
}
  • any,null,undefined,void: special types
  • null,undefined same as any types

  Typing Functions

home.ts


export class HomePage { 
  constructor() {
  }
  // assign the result to an incorrect data type,
  // the compiler will throw an error.
  function sayHello(theName: string): string { 
    return 'Hello, '+theName; 
  }
  // Use :void for no return
  function log(message): void { 
    console.log(message); 
  }
}
  • must return string type
  • no return in void type
