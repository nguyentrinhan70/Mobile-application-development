Native plugin 'camera' trong Ionic

  Cài đặt

Command Prompt


// cài cordova plugin: camera 
ionic cordova plugin add cordova-plugin-camera

// cài ionic package: camera 
npm install --save @ionic-native/camera

app.module.ts


import { Camera } from '@ionic-native/camera'; // <--

@NgModule({
  declarations: [
    MyApp,
  ],
  providers: [
    Camera // <--
  ]  
})
export class AppModule {}
